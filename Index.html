<html>

<head>

  <meta charset="UTF-8">

  <title>View DSF Info</title>

  <style>

    body { font-family: sans-serif; padding: 2rem; max-width: 600px; margin: auto; }

    .not-found { color: red; }

  </style>

</head>

<body>

  <h2>DREAM Scholarship Fund</h2>

  <div id="output">Loading...</div>

  <script>

    const sheetURL = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vRP-JEk4Rl8tHkl_U8NXxegpsHOuSbSkmtBCCCj54arskY-qzGnPSeW9HU92aYOzC3SeBSa9-Kh86hq/pub?output=tsv';

   

    const params = new URLSearchParams(window.location.search);

    const id = params.get('id');

    fetch(sheetURL)

      .then(response => response.text())

      .then(data => {

        const lines = data.trim().split('\n');

        const headers = lines[0].split('\t');

        const rows = lines.slice(1).map(line => {

          const values = line.split('\t');

          return headers.reduce((obj, key, i) => {

            obj[key] = values[i];

            return obj;

          }, {});

        });

        const match = rows.find(row => row.ID === id);

        if (match) {

          document.getElementById('output').innerHTML = `

            <p><strong>Child Name:</strong> ${match.Name}</p>

            <p><strong>Team:</strong> ${match.Team}</p>

            <p><strong>Total Raised:</strong> ${match['Total Raised']}</p>

            <p><strong>Region:</strong> ${match.Region}</p>

          `;

        } else {

          document.getElementById('output').innerHTML = `<p class="not-found">No record found for ID: ${id}</p>`;

        }

      })

      .catch(err => {

        document.getElementById('output').innerHTML = `<p class="not-found">Error loading data</p>`;

      });

  </script>

</body>

</html>
